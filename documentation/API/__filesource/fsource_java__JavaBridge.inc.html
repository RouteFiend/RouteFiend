  373.    * // available to Java. But only if it has been requested by Java:
  374.    * java_call_with_continuation();

  380.    * functions or methods, to debug or test certain
  381.    * functionality from Java while keeping the script running in a
  382.    * production environment. The Java code follows:

  385.    * import java.net.*;
  386.    * import java.io.*;
  387.    * class SalesTest {

  393.    *         Object res=e.eval(new php.java.script.URLReader(new URL("http://localhost/calculateSales.php")));

  395.    *         ((Closeable)e).close();
