  544.     private function parse ($type) {
  545.       if (PDB_DEBUG) pdb_Logger::debug("parse:::$type");
  546.  
  547.       $this->beginStatement = true;
  548.       $pLevel = 0;
  549.  
  550.       do {
  551.         $token = current($this->code);
  552.         if (!is_array($token)) {
  553.           if (PDB_DEBUG) pdb_Logger::debug(":::".$token);
  554.           if (!$pLevel && $type==self::FUNCTION_BLOCK && $token=='}') $this->writeStep($pLevel);
  555.           $this->write($token);
  556.           if ($this->inPhp && !$this->inDQuote) {
  557.             $this->beginStatement = false; 
  558.             switch($token) {
  559.             case '(': 
  560.               $pLevel++;
  561.               break;
  562.             case ')':
  563.               if (!--$pLevel && $type==self::EXPRESSION) return;
  564.               break;
  565.             case '{': 
  566.               $this->next();
  567.               $this->parseBlock(); 
  568.               break;
  569.             case '}': 
  570.               if (!$pLevel) return;
  571.               break;
  572.             case ';':
  573.               if (!$pLevel) {
  574.                 if ($type==self::STATEMENT) return;
  575.                 $this->beginStatement = true; 
  576.               }
  577.               break;
  578.             }
  579.           }
  580.         } else {
  581.           if (PDB_DEBUG) pdb_Logger::debug(":::".$token[1].":(".token_name($token[0]).')');
  582.  
  583.           if ($this->inDQuote) {
  584.             $this->write($token[1]);
  585.             continue;
  586.           }
  587.  
  588.           switch($token[0]) {
  589.  
  590.           case T_OPEN_TAG: 
  591.           case T_START_HEREDOC:
  592.           case T_OPEN_TAG_WITH_ECHO: 
  593.             $this->beginStatement = $this->inPhp = true;
  594.             $this->write($token[1]);
  595.             break;
  596.  
  597.           case T_END_HEREDOC:
  598.           case T_CLOSE_TAG: 
  599.             $this->writeStep($pLevel);
  600.  
  601.             $this->write($token[1]);
  602.             $this->beginStatement = $this->inPhp = false; 
  603.             break;
  604.  
  605.           case T_FUNCTION:
  606.             $this->write($token[1]);
  607.             $this->writeCall();
  608.             $this->next();
  609.             $this->parseFunction();
  610.             $this->beginStatement = true;
  611.             break;
  612.  
  613.           case T_ELSE:
  614.             $this->write($token[1]);
  615.             if ($this->nextIs('{')) {
  616.               $this->writeNext();
  617.               $this->next();
  618.  
  619.               $this->parseBlock();
  620.             } else {
  621.               $this->next();
  622.  
  623.               /* create an artificial block */
  624.               $this->write('{');
  625.               $this->beginStatement = true;
  626.               $this->writeStep($pLevel);
  627.               $this->parseStatement();
  628.               $this->write('}');
  629.  
  630.             }
  631.             if ($type==self::STATEMENT) return;
  632.  
  633.             $this->beginStatement = true;
  634.             break;
  635.  
  636.           case T_DO:
  637.             $this->writeStep($pLevel);
  638.             $this->write($token[1]);
  639.             if ($this->nextIs('{')) {
  640.               $this->writeNext();
  641.               $this->next();
  642.  
  643.               $this->parseBlock();
  644.               $this->next();
  645.  
  646.             } else {
  647.               $this->next();
  648.  
  649.               /* create an artificial block */
  650.               $this->write('{');
  651.               $this->beginStatement = true;
  652.               $this->writeStep($pLevel);
  653.               $this->parseStatement();
  654.               $this->next();
  655.               $this->write('}');
  656.             }
  657.             $token = current($this->code);
  658.             $this->write($token[1]);
  659.  
  660.             if ($token[0]!=T_WHILE) trigger_error("parse error", E_USER_ERROR);
  661.             $this->next();
  662.             $this->parseExpression();
  663.  
  664.             if ($type==self::STATEMENT) return;
  665.  
  666.             $this->beginStatement = true;
  667.             break;
  668.  
  669.           case T_CATCH:
  670.           case T_IF:
  671.           case T_ELSEIF:
  672.           case T_FOR:
  673.           case T_FOREACH:
  674.           case T_WHILE:
  675.             $this->writeStep($pLevel);
  676.  
  677.             $this->write($token[1]);
  678.             $this->next();
  679.  
  680.             $this->parseExpression();
  681.  
  682.             if ($this->nextIs('{')) {
  683.               $this->writeNext();
  684.               $this->next();
  685.  
  686.               $this->parseBlock();
  687.  
  688.  
  689.             } else {
  690.               $this->next();
  691.               /* create an artificial block */
  692.               $this->write('{');
  693.               $this->beginStatement = true;
  694.               $this->writeStep($pLevel);
  695.               $this->parseStatement();
  696.               $this->write('}');
  697.             }
  698.  
  699.             if ($this->nextTokenIs(array(T_ELSE, T_ELSEIF, T_CATCH))) {
  700.               $this->beginStatement = false;
  701.             } else {
  702.               if ($type==self::STATEMENT) return;
  703.               $this->beginStatement = true;
  704.             }
  705.             break;
  706.  
  707.           case T_REQUIRE_ONCE:
  708.           case T_INCLUDE_ONCE: 
  709.           case T_INCLUDE: 
  710.           case T_REQUIRE: 
  711.             $this->writeStep($pLevel);
  712.             $this->writeInclude((($token[0]==T_REQUIRE_ONCE) || ($token[0]==T_INCLUDE_ONCE)) ? 1 : 0);
  713.  
  714.             if ($type==self::STATEMENT) return;
  715.  
  716.             $this->beginStatement = true;
  717.             break;
  718.  
  719.           case T_CLASS:
  720.             $this->write($token[1]);
  721.             $this->writeNext();
  722.             if ($this->nextIs('{')) {
  723.               $this->writeNext();
  724.               $this->next();
  725.               $this->parseBlock(); 
  726.               $this->beginStatement = true;
  727.             } else {
  728.               $this->writeNext();
  729.               $this->beginStatement = false;
  730.             }
  731.             break;
  732.  
  733.           case T_CASE:
  734.           case T_DEFAULT:
  735.           case T_PUBLIC:
  736.           case T_PRIVATE:
  737.           case T_PROTECTED:
  738.           case T_STATIC:
  739.           case T_CONST:
  740.           case T_GLOBAL:
  741.           case T_ABSTRACT:
  742.             $this->write($token[1]);
  743.             $this->beginStatement = false;
  744.             break;
  745.  
  746.           default:
  747.             $this->writeStep($pLevel);
  748.             $this->write($token[1]);
  749.             $this->beginStatement = false;
  750.             break;
  751.     
  752.           }
  753.         }
  754.       } while($this->each());
  755.     }

  768.     }

  1526.   public function step($scriptName, $line, $vars) {
