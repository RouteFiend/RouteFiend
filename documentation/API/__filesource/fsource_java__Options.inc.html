  31. /**
  32.  * Helper function: Extract the URL from base and
  33.  * and set host, port and servlet accordingly.
  34.  * For example when the user has called:
  35.  * require_once("http://localhost:8080/JavaBridge/java/Java.inc");
  36.  * the JAVA_HOSTS is set to localhost:8080 and
  37.  * JAVA_SERVLET to /JavaBridge/JavaBridge.phpjavabridge.
  38.  * @access private
  39.  */
  40. function java_defineHostFromInitialQuery($java_base) {
  41.   if($java_base!="java") {
  42.     $url = parse_url($java_base);
  43.     if(isset($url["scheme"]) && ($url["scheme"]=="http" || $url["scheme"]=="https")) {
  44.       $scheme = $url["scheme"] == "https" ? "ssl://" : "";
  45.       $host = $url["host"];
  46.       $port = $url["port"];
  47.       $path = $url["path"];
  48.       define ("JAVA_HOSTS", "${scheme}${host}:${port}");
  49.       $dir = dirname($path);
  50.       define ("JAVA_SERVLET", "$dir/servlet.phpjavabridge"); // On ;; On or User
  51.       return true;
  52.     }
  53.   }
  54.   return false;
  55. }
