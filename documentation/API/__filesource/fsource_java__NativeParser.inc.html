  50. class java_NativeParser {
  51.   public $parser, $handler;
  52.   public $level, $event;

  54.   public $buf;

  56.   function java_NativeParser($handler) {
  57.     $this->handler = $handler;
  58.     $this->parser = xml_parser_create();
  59.     xml_parser_set_option($this->parser, XML_OPTION_CASE_FOLDING, 0);
  60.     xml_set_object($this->parser, $this);
  61.     xml_set_element_handler($this->parser, "begin", "end");
  62.     xml_parse($this->parser, "<F>");
  63.     $this->level = 0;
  64.   }
  65.   function begin($parser, $name, $param) {
  66.       
  67.     $this->event = true;
  68.     switch($name) {
  69.     case 'X': case 'A': $this->level+=1;
  70.     }
  71.     $this->handler->begin($name, $param);
  72.   }
  73.   function end($parser, $name) {
  74.       
  75.     $this->handler->end($name);
  76.     switch($name) {
  77.     case 'X': case 'A': $this->level-=1;
  78.     }
  79.   }
  80.   function getData($str) {
  81.     return base64_decode($str);
  82.   }
  83.   function parse() {
  84.     // NOTE: RECV_SIZE must be the same as the underlying 
  85.     // PHP stream buffer size, otherwise this function will hang.
  86.     do {
  87.       $this->event = false;
  88.       $buf = $this->buf = $this->handler->read(JAVA_RECV_SIZE); 
  89.       $len = strlen($buf);
  90.       if(!xml_parse($this->parser, $buf, $len==0)) {
  91.         $this->handler->protocol->handler->shutdownBrokenConnection(
  92.                sprintf("protocol error: %s, %s at col %d. Check the back end log for OutOfMemoryErrors.",
  93.                     $buf,
  94.                     xml_error_string(xml_get_error_code($this->parser)),
  95.                     xml_get_current_column_number($this->parser)));
  96.       }
  97.     } while(!$this->event || $this->level>0);
  98.   }
  99.  
  100.   function parserError() {
  101.     $this->handler->protocol->handler->shutdownBrokenConnection(
  102.        sprintf("protocol error: %s. Check the back end log for details.", $this->buf));
  103.   }
  104. }
