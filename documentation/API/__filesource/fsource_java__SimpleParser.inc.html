  73. class java_SimpleParser {
  74.   public $SLEN=256; // initial length of the parser string
  75.  
  76.   public $handler;
  77.   
  78.   public $tag, $buf, $len, $s;
  79.   public $type;
  80.  
  81.   function java_SimpleParser($handler) {
  82.     $this->handler = $handler;
  83.     $this->tag = array(new java_ParserTag(), new java_ParserTag(), new java_ParserTag());
  84.     $this->len = $this->SLEN;
  85.     $this->s = str_repeat(" ", $this->SLEN);
  86.     $this->type = $this->VOJD;
  87.   }
  88.   
  89.   // VOJD is VOID for f... windows (VOID is in winsock2.h)
  90.   public $BEGIN=0, $KEY=1, $VAL=2, $ENTITY=3, $VOJD=5, $END=6; 
  91.   public $level=0, $eor=0; public $in_dquote, $eot=false;
  92.   public $pos=0, $c=0, $i=0, $i0=0, $e;
  93.  
  94.   function RESET() {
  95.     $this->type=$this->VOJD;
  96.     $this->level=0;
  97.     $this->eor=0;
  98.     $this->in_dquote=false;
  99.     $this->i=0;
  100.     $this->i0=0;
  101.   }
  102.     
  103.   function APPEND($c) {
  104.     if($this->i>=$this->len-1) {
  105.       $this->s=str_repeat($this->s,2);
  106.       $this->len*=2;
  107.     } 
  108.     $this->s[$this->i++]=$c; 
  109.   }
  110.   function CALL_BEGIN() {
  111.     $pt=&$this->tag[1]->strings;
  112.     $st=&$this->tag[2]->strings;
  113.     $t=&$this->tag[0]->strings[0];
  114.     $name=$t->string[$t->off];
  115.     $n = $this->tag[2]->n;
  116.     $ar = array();
  117.     for($i=0; $i<$n; $i++) {
  118.       $ar[$pt[$i]->getString()] = $st[$i]->getString();
  119.     }
  120.     $this->handler->begin($name, $ar);
  121.   }
  122.   function CALL_END() {
  123.     $t=&$this->tag[0]->strings[0];
  124.     $name=$t->string[$t->off];
  125.     $this->handler->end($name);
  126.   }
  127.   function PUSH($t) { 
  128.     $str = &$this->tag[$t]->strings;
  129.     $n = &$this->tag[$t]->n;
  130.     $this->s[$this->i]='|';
  131.     if(!isset($str[$n])){$h=$this->handler; $str[$n]=$h->createParserString();}
  132.     $str[$n]->string=&$this->s;
  133.     $str[$n]->off=$this->i0;
  134.     $str[$n]->length=$this->i-$this->i0;
  135.     ++$this->tag[$t]->n;
  136.     $this->APPEND('|');
  137.     $this->i0=$this->i;
  138.   }
  139.   function parse() {
  140.     while($this->eor==0) {
  141.       if($this->c>=$this->pos) { 
  142.         
  143.         $this->buf=$this->handler->read(JAVA_RECV_SIZE); 
  144.         if(is_null($this->buf) || strlen($this->buf) == 0) 
  145.           $this->handler->protocol->handler->shutdownBrokenConnection("protocol error. Check the back end log for OutOfMemoryErrors.");
  146.         $this->pos=strlen($this->buf);
  147.         if($this->pos==0) break;
  148.         $this->c=0; 
  149.       }
  150.       switch(($ch=$this->buf[$this->c])) 
  151.         {/* --- This block must be compilable with an ansi C compiler or javac --- */
  152.         case '<': if($this->in_dquote) {$this->APPEND($ch); break;}
  153.           $this->level+=1;
  154.           $this->type=$this->BEGIN;
  155.           break;
  156.         case '\t': case '\f': case '\n': case '\r': case ' ': if($this->in_dquote) {$this->APPEND($ch); break;}
  157.           if($this->type==$this->BEGIN) {
  158.             $this->PUSH($this->type); 
  159.             $this->type = $this->KEY; 
  160.           }
  161.           break;
  162.         case '=': if($this->in_dquote) {$this->APPEND($ch); break;}
  163.           $this->PUSH($this->type);
  164.           $this->type=$this->VAL;
  165.           break;
  166.         case '/': if($this->in_dquote) {$this->APPEND($ch); break;}
  167.           if($this->type==$this->BEGIN) { $this->type=$this->END; $this->level-=1; }
  168.           $this->level-=1;
  169.           $this->eot=true; // used for debugging only
  170.           break;
  171.         case '>': if($this->in_dquote) {$this->APPEND($ch); break;}
  172.           if($this->type==$this->END){
  173.             $this->PUSH($this->BEGIN);
  174.             $this->CALL_END();
  175.           } else {
  176.             if($this->type==$this->VAL) $this->PUSH($this->type);
  177.             $this->CALL_BEGIN();
  178.           }
  179.           $this->tag[0]->n=$this->tag[1]->n=$this->tag[2]->n=0; $this->i0=$this->i=0;              /* RESET */
  180.           $this->type=$this->VOJD;
  181.           if($this->level==0) $this->eor=1; 
  182.           break;
  183.         case ';':
  184.           if($this->type==$this->ENTITY) {
  185.             switch ($this->s[$this->e+1]) {
  186.             case 'l': $this->s[$this->e]='<'; $this->i=$this->e+1; break; /* lt */
  187.             case 'g': $this->s[$this->e]='>'; $this->i=$this->e+1; break; /* gt */
  188.             case 'a': $this->s[$this->e]=($this->s[$this->e+2]=='m'?'&':'\''); $this->i=$this->e+1; break; /* amp, apos */
  189.             case 'q': $this->s[$this->e]='"'; $this->i=$this->e+1; break; /* quot */
  190.             default: $this->APPEND($ch);
  191.             }
  192.             $this->type=$this->VAL; //& escapes may only appear in values
  193.           } else {
  194.             $this->APPEND($ch);
  195.           }
  196.           break;
  197.         case '&': 
  198.           $this->type = $this->ENTITY;
  199.           $this->e=$this->i;
  200.           $this->APPEND($ch);
  201.           break;
  202.         case '"':
  203.           $this->in_dquote = !$this->in_dquote;
  204.           if(!$this->in_dquote && $this->type==$this->VAL) {
  205.             $this->PUSH($this->type);
  206.             $this->type = $this->KEY;
  207.           }
  208.           break;
  209.         default:
  210.           $this->APPEND($ch);
  211.         } /* ------------------ End of ansi C block ---------------- */
  212.       $this->c+=1;
  213.     }
  214.        $this->RESET();
  215.   }
