  167.     private function nextIs($chr) {

  242.     private function parse ($type) {
  243.       pdb_Logger::debug("parse:::$type");
  244.  
  245.       $this->beginStatement = true;
  246.       $pLevel = 0;
  247.  
  248.       do {
  249.         $token = current($this->code);
  250.         if (!is_array($token)) {
  251.           pdb_Logger::debug(":::".$token);
  252.           if (!$pLevel && $type==self::FUNCTION_BLOCK && $token=='}') $this->writeStep($pLevel);
  253.           $this->write($token);
  254.           if ($this->inPhp && !$this->inDQuote) {
  255.             $this->beginStatement = false; 
  256.             switch($token) {
  257.             case '(': 
  258.               $pLevel++;
  259.               break;
  260.             case ')':
  261.               if (!--$pLevel && $type==self::EXPRESSION) return;
  262.               break;
  263.             case '{': 
  264.               $this->next();
  265.               $this->parseBlock(); 
  266.               break;
  267.             case '}': 
  268.               if (!$pLevel) return;
  269.               break;
  270.             case ';':
  271.               if (!$pLevel) {
  272.                 if ($type==self::STATEMENT) return;
  273.                 $this->beginStatement = true; 
  274.               }
  275.               break;
  276.             }
  277.           }
  278.         } else {
  279.           pdb_Logger::debug(":::".$token[1].":(".token_name($token[0]).')');
  280.  
  281.           if ($this->inDQuote) {
  282.             $this->write($token[1]);
  283.             continue;
  284.           }
  285.  
  286.           switch($token[0]) {
  287.  
  288.           case T_OPEN_TAG: 
  289.           case T_START_HEREDOC:
  290.           case T_OPEN_TAG_WITH_ECHO: 
  291.             $this->beginStatement = $this->inPhp = true;
  292.             $this->write($token[1]);
  293.             break;
  294.  
  295.           case T_END_HEREDOC:
  296.           case T_CLOSE_TAG: 
  297.             $this->writeStep($pLevel);
  298.  
  299.             $this->write($token[1]);
  300.             $this->beginStatement = $this->inPhp = false; 
  301.             break;
  302.  
  303.           case T_FUNCTION:
  304.             $this->write($token[1]);
  305.             $this->writeCall();
  306.             $this->next();
  307.             $this->parseFunction();
  308.             $this->beginStatement = true;
  309.             break;
  310.  
  311.           case T_ELSE:
  312.             $this->write($token[1]);
  313.             if ($this->nextIs('{')) {
  314.               $this->writeNext();
  315.               $this->next();
  316.  
  317.               $this->parseBlock();
  318.             } else {
  319.               $this->next();
  320.  
  321.               /* create an artificial block */
  322.               $this->write('{');
  323.               $this->beginStatement = true;
  324.               $this->writeStep($pLevel);
  325.               $this->parseStatement();
  326.               $this->write('}');
  327.  
  328.             }
  329.             if ($type==self::STATEMENT) return;
  330.  
  331.             $this->beginStatement = true;
  332.             break;
  333.  
  334.           case T_DO:
  335.             $this->writeStep($pLevel);
  336.             $this->write($token[1]);
  337.             if ($this->nextIs('{')) {
  338.               $this->writeNext();
  339.               $this->next();
  340.  
  341.               $this->parseBlock();
  342.               $this->next();
  343.  
  344.             } else {
  345.               $this->next();
  346.  
  347.               /* create an artificial block */
  348.               $this->write('{');
  349.               $this->beginStatement = true;
  350.               $this->writeStep($pLevel);
  351.               $this->parseStatement();
  352.               $this->next();
  353.               $this->write('}');
  354.             }
  355.             $token = current($this->code);
  356.             $this->write($token[1]);
  357.  
  358.             if ($token[0]!=T_WHILE) trigger_error("parse error", E_USER_ERROR);
  359.             $this->next();
  360.             $this->parseExpression();
  361.  
  362.             if ($type==self::STATEMENT) return;
  363.  
  364.             $this->beginStatement = true;
  365.             break;
  366.  
  367.           case T_CATCH:
  368.           case T_IF:
  369.           case T_ELSEIF:
  370.           case T_FOR:
  371.           case T_FOREACH:
  372.           case T_WHILE:
  373.             $this->writeStep($pLevel);
  374.  
  375.             $this->write($token[1]);
  376.             $this->next();
  377.  
  378.             $this->parseExpression();
  379.  
  380.             if ($this->nextIs('{')) {
  381.               $this->writeNext();
  382.               $this->next();
  383.  
  384.               $this->parseBlock();
  385.  
  386.  
  387.             } else {
  388.               $this->next();
  389.               /* create an artificial block */
  390.               $this->write('{');
  391.               $this->beginStatement = true;
  392.               $this->writeStep($pLevel);
  393.               $this->parseStatement();
  394.               $this->write('}');
  395.             }
  396.  
  397.             if ($this->nextTokenIs(array(T_ELSE, T_ELSEIF, T_CATCH))) {
  398.               $this->beginStatement = false;
  399.             } else {
  400.               if ($type==self::STATEMENT) return;
  401.               $this->beginStatement = true;
  402.             }
  403.             break;
  404.  
  405.           case T_REQUIRE_ONCE:
  406.           case T_INCLUDE_ONCE: 
  407.           case T_INCLUDE: 
  408.           case T_REQUIRE: 
  409.             $this->writeStep($pLevel);
  410.             $this->writeInclude((($token[0]==T_REQUIRE_ONCE) || ($token[0]==T_INCLUDE_ONCE)) ? 1 : 0);
  411.  
  412.             if ($type==self::STATEMENT) return;
  413.  
  414.             $this->beginStatement = true;
  415.             break;
  416.  
  417.           case T_CLASS:
  418.             $this->write($token[1]);
  419.             $this->writeNext();
  420.             if ($this->nextIs('{')) {
  421.               $this->writeNext();
  422.               $this->next();
  423.               $this->parseBlock(); 
  424.               $this->beginStatement = true;
  425.             } else {
  426.               $this->writeNext();
  427.               $this->beginStatement = false;
  428.             }
  429.             break;
  430.  
  431.           case T_CASE:
  432.           case T_DEFAULT:
  433.           case T_PUBLIC:
  434.           case T_PRIVATE:
  435.           case T_PROTECTED:
  436.           case T_STATIC:
  437.           case T_CONST:
  438.           case T_GLOBAL:
  439.           case T_ABSTRACT:
  440.             $this->write($token[1]);
  441.             $this->beginStatement = false;
  442.             break;
  443.  
  444.           default:
  445.             $this->writeStep($pLevel);
  446.             $this->write($token[1]);
  447.             $this->beginStatement = false;
  448.             break;
  449.     
  450.           }
  451.         }
  452.       } while($this->each());
  453.     }

  455.     /**

  672. class pdb_Serializer {
  673.   private $serial;
  674.   private $depth;
  675.  
  676.   private function doSerialize ($o, $depth) {
  677.     $serial = &$this->serial;
  678.  
  679.     switch(gettype($o)) {
  680.     case 'object':
  681.       $serial.="O:";
  682.       $serial.=strlen(get_class($o));
  683.       $serial.=":\"";
  684.       $serial.=get_class($o);
  685.       $serial.="\":";
  686.       $serial.=count((array)$o);
  687.  
  688.       if ($depth <= $this->depth) {
  689.         $serial.=":{";
  690.         foreach((array)$o as $k=>$v) {
  691.           $serial.=serialize($k);
  692.           $this->doSerialize($v, $depth+1);
  693.         }
  694.         $serial.="}";
  695.       } else {
  696.         $serial .= ";";
  697.       }
  698.       break;
  699.  
  700.     case 'array':
  701.       $serial.="a:";
  702.       $serial.=count($o);
  703.  
  704.       if ($depth <= $this->depth) {
  705.         $serial.=":{";
  706.         foreach($o as $k=>$v) {
  707.           $serial.=serialize($k);
  708.           $this->doSerialize($v, $depth+1);
  709.         }
  710.         $serial.="}";
  711.       } else {
  712.         $serial.=";";
  713.       }
  714.       break;
  715.     default:
  716.       $serial.=serialize($o);
  717.       break;
  718.     }
  719.   }
  720.  
  721.   public function serialize ($obj, $depth) {
  722.     $this->serial = "";
  723.     $this->depth = $depth;
  724.

  727.     return $this->serial;

  2000.     foreach ($pdb_dbg->allFrames as $currentFrame)
  2001.       $currentFrame->stepNext = false;
  2002.  
  2003.     /* set new dynamic breakpoint */
  2004.     if ($stepNext != pdb_GoRequest::TYPE) {
  2005.       $currentFrame = $pdb_dbg->currentFrame;
  2006.

  2008.       if ($stepNext != pdb_StepOutRequest::TYPE)
  2009.         $currentFrame->stepNext = $stepNext;
  2010.

  2012.       while ($currentFrame = $currentFrame->parent) {
  2013.         $currentFrame->stepNext = $stepNext;
  2014.       }
  2015.     }
  2016.   }
