  322.   function handleRequests() {
  323.     $tail_call = false;
  324.     do {
  325.       $this->stack=array($this->arg=$this->simpleArg);
  326.       $this->idx = 0;
  327.       $this->parser->parse();
  328.  
  329.       /* pull off A, if any */
  330.       if((count($this->stack)) > 1) {
  331.         $arg = array_pop($this->stack);
  332.         $this->apply($arg);
  333.         $tail_call = true;            // we don't expect a result
  334.       } else {
  335.         $tail_call = false;
  336.       }
  337.  
  338.       $this->stack=null;
  339.     } while($tail_call);
  340.     return 1;
  341.   }

  381.   function begin($name, $st) {
  382.     $arg = $this->arg;
  383.     switch($name[0]) {
  384.     case 'A':                        /* receive apply args as normal array */
  385.       $object = $this->globalRef->get($this->getExact($st['v']));
  386.       $newArg = new java_ApplyArg($this, 'A',
  387.                                   $this->parser->getData($st['m']),
  388.                                   $this->parser->getData($st['p']),
  389.                                   $object,
  390.                                   $this->getExact($st['n']));
  391.       $this->link($arg, $newArg);
  392.       array_push($this->stack, $this->arg = $newArg);
  393.       break;
  394.     case 'X': 
  395.       $newArg = new java_CompositeArg($this, $st['t']);
  396.       $this->link($arg, $newArg);
  397.       array_push($this->stack, $this->arg = $newArg);
  398.       break;
  399.     case 'P':
  400.       if($arg->type=='H') { /* hash table */
  401.         $s = $st['t'];
  402.         if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult prepare hash:"; echo sprintf("%s", $st['t']); echo "\n";}
  403.         if($s[0]=='N') { /* number */
  404.           $arg->setIndex($this->getExact($st['v']));
  405.           if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult array: index:"; echo sprintf("%s", $st['v']); echo "\n";}
  406.         } else {
  407.           $arg->setIndex($this->parser->getData($st['v']));
  408.           if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult hash: index:"; echo sprintf("%s", $this->parser->getData($st['v'])); echo "\n";}
  409.         }
  410.       } else {                    /* array */
  411.         $arg->setNextIndex();
  412.       }
  413.       break;
  414.     case 'S':
  415.       $arg->setResult($this->parser->getData($st['v']));
  416.       if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult string:"; echo sprintf("%s", $this->parser->getData($st['v'])); echo "\n";}
  417.       break;
  418.     case 'B':
  419.       $s=$st['v'];
  420.       $arg->setResult($s[0]=='T');
  421.       if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult bool:"; echo sprintf("%s", $st['v']); echo "\n";}
  422.       break;
  423.     case 'L':                    // unsigned long
  424.       $sign = $st['p'];
  425.       $val = $this->getExact($st['v']);
  426.       if($sign[0]=='A') $val*=-1;
  427.       $arg->setResult($val);
  428.       if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult long:"; echo sprintf("%s, sign: %s", $st['v'], $st['p']); echo "\n";}
  429.       break;
  430.     case 'D':
  431.       $arg->setResult($this->getInexact($st['v']));
  432.       if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult double:"; echo sprintf("%s", $st['v']); echo "\n";}
  433.       break;
  434.     case 'V':
  435.       if ($st['n']!='T') {
  436.         if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult VOID:"; echo "\n";}
  437.         $arg->setVoidSignature();
  438.       }
  439.       // fall through
  440.     case 'N':
  441.       $arg->setResult(null);
  442.       if(<a href="../java/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult null\n"; }
